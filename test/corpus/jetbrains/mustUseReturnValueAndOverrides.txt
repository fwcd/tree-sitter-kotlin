interface Base1 {

@MustUseReturnValues
interface Base2 {
    fun mustUse(): String
    @IgnorableReturnValue fun ignorable(): String
}

class Impl2: Base2 {

    override fun ignorable(): String = ""

class Impl3: Base2 {
    override fun mustUse(): String = ""
    @IgnorableReturnValue override fun ignorable(): String = ""
}
